<?php
require 'db.php';

$message = '';

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $username = trim($_POST['username']);
    $password = $_POST['password'];
    $role = $_POST['role']; // 'admin' hoặc 'viewer'

    // 1. Kiểm tra dữ liệu
    if (empty($username) || empty($password)) {
        $message = '<div class="alert alert-danger">Vui lòng nhập đầy đủ thông tin!</div>';
    } else {
        // 2. Kiểm tra user đã tồn tại chưa
        $stmt = $pdo->prepare("SELECT id FROM users WHERE username = ?");
        $stmt->execute([$username]);
        if ($stmt->fetch()) {
            $message = '<div class="alert alert-danger">Tên đăng nhập này đã tồn tại!</div>';
        } else {
            // 3. Mã hóa mật khẩu
            $hashedPassword = password_hash($password, PASSWORD_DEFAULT);

            // 4. Thêm vào database
            try {
                $sql = "INSERT INTO users (username, password, role) VALUES (?, ?, ?)";
                $pdo->prepare($sql)->execute([$username, $hashedPassword, $role]);
                $message = '<div class="alert alert-success">Đã tạo tài khoản <b>' . htmlspecialchars($username) . '</b> (Quyền: ' . $role . ') thành công! <a href="login.php">Đăng nhập ngay</a></div>';
            } catch (Exception $e) {
                $message = '<div class="alert alert-danger">Lỗi: ' . $e->getMessage() . '</div>';
            }
        }
    }
}
?>

<!DOCTYPE html>
<html lang="vi">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Tạo tài khoản</title>
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/css/bootstrap.min.css" rel="stylesheet">
</head>

<body class="bg-light d-flex align-items-center justify-content-center" style="min-height: 100vh;">
    <div class="card shadow p-4" style="max-width: 400px; width: 100%;">
        <h3 class="text-center mb-4 text-success fw-bold">Tạo Tài Khoản Mới</h3>

        <?= $message ?>

        <form method="POST">
            <div class="mb-3">
                <label class="form-label">Tên đăng nhập</label>
                <input type="text" name="username" class="form-control" required placeholder="Ví dụ: admin2">
            </div>
            <div class="mb-3">
                <label class="form-label">Mật khẩu</label>
                <input type="password" name="password" class="form-control" required placeholder="Nhập mật khẩu...">
            </div>
            <div class="mb-3">
                <label class="form-label">Quyền hạn</label>
                <select name="role" class="form-select">
                    <option value="viewer">Viewer (Chỉ xem)</option>
                    <option value="admin">Admin (Toàn quyền)</option>
                </select>
            </div>
            <button type="submit" class="btn btn-success w-100 py-2 fw-bold">Tạo tài khoản</button>
        </form>

        <div class="text-center mt-3 border-top pt-3">
            <a href="login.php" class="text-decoration-none me-3">Đăng nhập</a>
            <a href="index.php" class="text-decoration-none">Về trang chủ</a>
        </div>
    </div>
</body>

</html>
